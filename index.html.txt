                <li><strong>Blaze Tozu:</strong> İksirlerin temel bileşeni. Blaze Rod'larından elde edilir.</li>
                <li><strong>Ağaç Meyvesi:</strong> İksirlerin etkisini artırmak için kullanılan malzemedir.</li>

                <li><strong>Gölge Tozu:</strong> Görünmezlik iksirlerinin yapımında kullanılır.</li>
            </ul>

    <script>
        // Basit bir yorum ekleme işlevi
        document.getElementById('comment-form').addEventListener('submit', function(event) {
            event.preventDefault();

            var name = document.getElementById('name').value;
            var comment = document.getElementById('comment').value;

            if (name && comment) {
                var commentsList = document.getElementById('comments-list');
                var newComment = document.createElement('li');
                newComment.innerHTML = '<strong>' + name + ':</strong> ' + comment;
                commentsList.appendChild(newComment);

                // Formu temizle
                document.getElementById('comment-form').reset();
            }
        });
